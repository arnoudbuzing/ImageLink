9/// Internal utility to convert a NumericArray to a grayscale ImageBuffer.
10fn to_luma(array: &NumericArray<u8>) -> Option<ImageBuffer<image::Luma<u8>, Vec<u8>>> {
11    let dims = array.dimensions();
12    if dims.len() < 2 { return None; }
13    let h = dims[0] as u32;
14    let w = dims[1] as u32;
15    let ch = if dims.len() == 3 { dims[2] as u32 } else { 1 };
16    let s = array.as_slice().to_vec();
17    Some(if ch == 3 {
18        let img = ImageBuffer::<Rgb<u8>, _>::from_raw(w, h, s)?;
19        imageproc::map::map_pixels(&img, |p| image::Luma([((p.0[0] as u32 + p.0[1] as u32 + p.0[2] as u32) / 3) as u8]))
20    } else if ch == 4 {
21        let img = ImageBuffer::<Rgba<u8>, _>::from_raw(w, h, s)?;
22        imageproc::map::map_pixels(&img, |p| image::Luma([((p.0[0] as u32 + p.0[1] as u32 + p.0[2] as u32) / 3) as u8]))
23    } else {
24        ImageBuffer::<image::Luma<u8>, _>::from_raw(w, h, s)?
25    })
26}

34/// Returns a flat array of shape [channels × 256] (row-major) with per-channel histograms.
35#[export]
36pub fn histogram_memory(array: &NumericArray<u8>) -> NumericArray<u32> {
37    let dims = array.dimensions();
38    if dims.len() < 2 { return NumericArray::<u32>::from_slice(&[0u32; 256]); }
39    let h = dims[0] as u32;
40    let w = dims[1] as u32;
41    let channels = if dims.len() == 3 { dims[2] as u32 } else { 1 };
42    let slice = array.as_slice();
43    let mut result: Vec<u32> = Vec::with_capacity((channels as usize) * 256);
44    if channels == 3 {
45        let img = ImageBuffer::<Rgb<u8>, _>::from_raw(w, h, slice.to_vec()).unwrap();
46        for i in 0..3usize {
47            let ch = imageproc::map::map_pixels(&img, |p| image::Luma([p.0[i]]));
48            result.extend_from_slice(&imageproc::stats::histogram(&ch).channels[0]);
49        }
50    } else if channels == 4 {
51        let img = ImageBuffer::<Rgba<u8>, _>::from_raw(w, h, slice.to_vec()).unwrap();
52        for i in 0..4usize {
53            let ch = imageproc::map::map_pixels(&img, |p| image::Luma([p.0[i]]));
54            result.extend_from_slice(&imageproc::stats::histogram(&ch).channels[0]);
55        }
56    } else {
57        let img = ImageBuffer::<image::Luma<u8>, _>::from_raw(w, h, slice.to_vec()).unwrap();
58        result.extend_from_slice(&imageproc::stats::histogram(&img).channels[0]);
59    }
60    NumericArray::<u32>::from_slice(&result)
61}

63/// Peak Signal-to-Noise Ratio between two same-size images.
64#[export]
65pub fn psnr_memory(array1: &NumericArray<u8>, array2: &NumericArray<u8>) -> f64 {
66    let d1 = array1.dimensions();
67    let d2 = array2.dimensions();
68    if d1 != d2 || d1.len() < 2 { return 0.0; }
69    let h = d1[0] as u32;
70    let w = d1[1] as u32;
71    let ch = if d1.len() == 3 { d1[2] as u32 } else { 1 };
72    if ch == 3 {
73        let i1 = ImageBuffer::<Rgb<u8>, _>::from_raw(w, h, array1.as_slice().to_vec()).unwrap();
74        let i2 = ImageBuffer::<Rgb<u8>, _>::from_raw(w, h, array2.as_slice().to_vec()).unwrap();
75        imageproc::stats::peak_signal_to_noise_ratio(&i1, &i2)
76    } else if ch == 4 {
77        let i1 = ImageBuffer::<Rgba<u8>, _>::from_raw(w, h, array1.as_slice().to_vec()).unwrap();
78        let i2 = ImageBuffer::<Rgba<u8>, _>::from_raw(w, h, array2.as_slice().to_vec()).unwrap();
79        imageproc::stats::peak_signal_to_noise_ratio(&i1, &i2)
80    } else {
81        let i1 = ImageBuffer::<image::Luma<u8>, _>::from_raw(w, h, array1.as_slice().to_vec()).unwrap();
82        let i2 = ImageBuffer::<image::Luma<u8>, _>::from_raw(w, h, array2.as_slice().to_vec()).unwrap();
83        imageproc::stats::peak_signal_to_noise_ratio(&i1, &i2)
84    }
85}

87/// 64-bit perceptual average-hash: resize to 8×8, compare each pixel to mean.
88#[export]
89pub fn phash_memory(array: &NumericArray<u8>) -> i64 {
90    let luma = match to_luma(array) { Some(l) => l, None => return 0 };
91    let small = image::imageops::resize(&luma, 8, 8, image::imageops::FilterType::Lanczos3);
92    let pixels: Vec<u32> = small.pixels().map(|p| p.0[0] as u32).collect();
93    if pixels.len() < 64 { return 0; }
94    let mean = pixels.iter().sum::<u32>() / 64;
95    pixels.iter().take(64).enumerate().fold(0u64, |acc, (i, &v)| {
96        if v > mean { acc | (1u64 << i) } else { acc }
97    }) as i64
98}

100/// Otsu's optimal binarisation threshold (0-255) on the luminance channel.
101#[export]
102pub fn otsu_level_memory(array: &NumericArray<u8>) -> i64 {
103    match to_luma(array) {
104        Some(luma) => imageproc::contrast::otsu_level(&luma) as i64,
105        None => 0,
106    }
107}

118/// Adjusts the luminance of an image to match the histogram of another image.
119#[export]
120pub fn match_histogram_memory(array1: &NumericArray<u8>, array2: &NumericArray<u8>) -> NumericArray<u8> {
121    let d1 = array1.dimensions();
122    let d2 = array2.dimensions();
123    if d1.len() < 2 || d2.len() < 2 {
124        return NumericArray::<u8>::from_slice(&[]);
125    }
126    
127    if let (Some(luma1), Some(luma2)) = (to_luma(array1), to_luma(array2)) {
128        let res = imageproc::contrast::match_histogram(&luma1, &luma2);
129        NumericArray::<u8>::from_array(&[d1[0], d1[1], 1], &res.into_raw())
130    } else {
131        NumericArray::<u8>::from_slice(&[])
132    }
